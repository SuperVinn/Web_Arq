Ejemplo práctico

Simulación de ejecución con instrucciones elementales y salida de datos.

contador_digitos.asm (EMU8086) 
; Programa para contar dígitos de un número en AX
.model small
.code
main proc
    MOV AX, 1234h   ; Número de prueba (4 dígitos hexadecimales)
    XOR CX, CX      ; Limpiar CX (actuará como nuestro contador)
    MOV BX, 10      ; Divisor base 10

contar_loop:
    XOR DX, DX      ; Limpiar DX para la división
    DIV BX          ; Dividir AX entre 10 (Cociente en AX, Residuo en DX)
    INC CX          ; Incrementamos el contador de dígitos
    CMP AX, 0       ; ¿El cociente ya es cero?
    JNE contar_loop ; Si no es cero, repetir el ciclo

    ; Al finalizar, el registro CX contiene el valor 4
    MOV AH, 4Ch
    INT 21h
main endp
end main
